class Solution {
public:
    int maxArea(vector<int>& height) {
        int n=height.size();
        int res=0;
        int left=0,right=n-1;
        while(left<right){
            int vol=(right-left)*(min(height[left], height[right]));
            res=max(vol,res);
            if (height[left] < height[right]) {
                left++;
            } else {
                right--;
            }
        }
        return res;
    }
};
