class Solution {
public:
    int minTimeToVisitAllPoints(vector<vector<int>>& points) {
        int time=0;
        for(int i=1;i<points.size();i++){
            int horizontal = abs(points[i][0]-points[i-1][0]);
            int vertical = abs(points[i][1]-points[i-1][1]);
            time+=max(vertical,horizontal);
        }
        return time;
    }
};
